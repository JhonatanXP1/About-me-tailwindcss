using System;
public class EjemploAleatorio
{
public static void Main( string[] args)
    {
      // Crear una instancia de la clase Random
      Random random = new Random();

      // Generar un número entero aleatorio entre 0 y 99
      int numeroAleatorio = random.Next(100);
      Console.WriteLine($"Número aleatorio entre 0 y 99: {numeroAleatorio}");

      // Generar un número entero aleatorio entre 1 y 100 (inclusive)
      int numeroAleatorioRango = random.Next(1, 101); // El segundo parámetro es exclusivo
      Console.WriteLine($"Número aleatorio entre 1 y 100: {numeroAleatorioRango}");

      // Generar un número de punto flotante aleatorio entre 0.0 y 1.0 (exclusivo)
        double numeroDecimalAleatorio = random.NextDouble();
        Console.WriteLine($"Número decimal aleatorio: {numeroDecimalAleatorio}");
    }
}
